<?php
session_start();
include 'connection/config.php';
include 'functions.php';

if (isset($_POST['verify'])) {
    $otp = $_POST['otp'];

    if ($otp == $_SESSION['otp']) {
        $user_data = $_SESSION['user_data'];
        $role = $user_data['role'];
        $user_id = $user_data['user_id'];

        // Common user insertion into tbl_user
        $insertUser = "INSERT INTO tbl_user (user_id, first_name, middle_name, last_name, email, username, password, mobile, address, role, is_logged_in) 
                       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 0)";
        $stmt = $conn->prepare($insertUser);
        $stmt->bind_param(
            "ssssssssss",
            $user_id,
            $user_data['first_name'],
            $user_data['middle_name'],
            $user_data['last_name'],
            $user_data['email'],
            $user_data['username'],
            $user_data['password'],
            $user_data['mobile'],
            $user_data['address'],
            $role
        );

        if ($stmt->execute()) {
            logActivity($user_id, $role, 'Registered in tbl_user');

            // Insert into `tbl_residents` if the role is `resident`
            if ($role === 'resident') {
                $insertResident = "INSERT INTO tbl_residents (user_id, first_name, middle_name, last_name, mobile, address) 
                                   VALUES (?, ?, ?, ?, ?, ?)";
                $stmt = $conn->prepare($insertResident);
                $stmt->bind_param(
                    "ssssss",
                    $user_id,
                    $user_data['first_name'],
                    $user_data['middle_name'],
                    $user_data['last_name'],
                    $user_data['mobile'],
                    $user_data['address']
                );
                $stmt->execute();
                logActivity($user_id, $role, 'Registered in tbl_residents');
            }

            // Insert into `tbl_brgyofficer` if the role is `barangay_official`
            elseif ($role === 'barangay_official') {
                $insertBrgyOfficer = "INSERT INTO tbl_brgyofficer (user_id, first_name, middle_name, last_name, mobile, address) 
                                      VALUES (?, ?, ?, ?, ?, ?)";
                $stmt = $conn->prepare($insertBrgyOfficer);
                $stmt->bind_param(
                    "ssssss",
                    $user_id,
                    $user_data['first_name'],
                    $user_data['middle_name'],
                    $user_data['last_name'],
                    $user_data['mobile'],
                    $user_data['address']
                );
                $stmt->execute();
                logActivity($user_id, $role, 'Registered in tbl_brgyofficer');
            }

            // Clear session data after successful registration
            unset($_SESSION['otp'], $_SESSION['user_data']);
            header('Location: login.php');
            exit();
        } else {
            $error[] = 'Registration failed. Try again.';
        }
    } else {
        $error[] = 'Invalid OTP.';
    }
}

        // OTP input handling
        const otpInputs = document.querySelectorAll('.otp-input');
        const otpHidden = document.getElementById('otp');

        otpInputs.forEach((input, index) => {
            input.addEventListener('keyup', (e) => {
                const value = e.target.value;
                
                if (value.length === 1 && index < otpInputs.length - 1) {
                    otpInputs[index + 1].focus();
                }

                // Combine all inputs into hidden field
                otpHidden.value = Array.from(otpInputs).map(input => input.value).join('');
            });

            input.addEventListener('keydown', (e) => {
                if (e.key === 'Backspace' && !e.target.value && index > 0) {
                    otpInputs[index - 1].focus();
                }
            });
        });

        // Resend timer
        let timeLeft = 60;
        const timerElement = document.getElementById('timer');
        
        const countdown = setInterval(() => {
            timeLeft--;
            timerElement.textContent = timeLeft;
            
            if (timeLeft <= 0) {
                clearInterval(countdown);
                document.querySelector('.resend-timer').innerHTML = 
                    '<a href="#" class="text-primary" onclick="resendOTP()">Resend verification code</a>';
            }
        }, 1000);
